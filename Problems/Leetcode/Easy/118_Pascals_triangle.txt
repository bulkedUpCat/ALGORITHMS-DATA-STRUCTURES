class Solution {
public:
    vector<vector<int>> generate(int numRows) {
        vector<vector<int>> dp(numRows);

        for (int i = 0; i < numRows; i++){
            vector<int> row(i + 1);
            row[0] = 1;
            row[i] = 1;
            dp[i] = row;
        }

        for (int i = 2; i < numRows; i++){
            for (int j = 1; j < i; j++){
                dp[i][j] = dp[i - 1][j - 1] + dp[i - 1][j];
            }
        }

        return dp;
    }
};
